class Solution(object):
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        even = [i for i in A if i % 2]
        odd = [i for i in A if not i % 2]

        return [i for n in zip(odd, even) for i in n]
